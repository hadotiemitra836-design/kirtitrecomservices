        /**
         * Calls the Gemini API to generate content.
         * Implements exponential backoff for retries.
         */
        const fetchWithGemini = async (userQuery, systemPrompt, maxRetries = 5) => {
            const payload = {
                contents: [{ parts: [{ text: userQuery }] }],
                systemInstruction: { parts: [{ text: systemPrompt }] },
            };

            for (let i = 0; i < maxRetries; i++) {
                try {
                    const response = await fetch(API_URL, {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify(payload)
                    });

                    if (response.ok) {
                        const result = await response.json();
                        const text = result.candidates?.[0]?.content?.parts?.[0]?.text;
                        return text || "Error: Could not extract generated text from response.";
                    } else if (response.status === 429 && i < maxRetries - 1) {
                        // Rate limit error (429) - wait and retry
                        const delay = Math.pow(2, i) * 1000 + Math.random() * 1000;
                        await new Promise(resolve => setTimeout(resolve, delay));
                        // Continue loop to retry
                    } else {
                        // Other non-retryable error
                        throw new Error(`API returned status ${response.status}: ${response.statusText}`);
                    }
                } catch (error) {
                    if (i === maxRetries - 1) {
                        console.error("Gemini API call failed after max retries:", error);
                        return `Error: Failed to connect to AI assistant. Please try again later. (${error.message})`;
                    }
                    // Handle network errors or JSON parsing errors (will be caught in the outer try-catch)
                    const delay = Math.pow(2, i) * 1000 + Math.random() * 1000;
                    await new Promise(resolve => setTimeout(resolve, delay));
                }
            }
            return "Error: Exceeded maximum retry attempts.";
        };
